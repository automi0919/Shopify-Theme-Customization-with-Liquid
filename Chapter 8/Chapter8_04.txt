const addCart = (el) => {
  let formData = {
    'items': [
      {
        id: el.previousElementSibling.value
      }
    ]
  };

  fetch('/cart/add.js', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(formData)
  })
  .then(success => {
    console.log("Success:", success);
  })
  .catch((error) => {
    console.error('Error:', error);
  });
}
